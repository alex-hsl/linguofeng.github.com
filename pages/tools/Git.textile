<section>
    <div class="page-header">
        <h3>二、常用命令</h3>
    </div>
    
    <pre>
       $ git config --global user.name "linguofeng"
       $ git config --global user.email linguofeng@msn.com
       $ git init                // 初始化一个Git版本库
       $ git add                 // 添加文件到版本库（未正式添加）
       $ git commit -m 'message' // 提交文件到版本库
    </pre>
</section>
